129  VectorXi elected(n_trees * max_leaf_size);

136  if(density < 1) {

196  int elect(const VectorXf &projected_query, int k, int votes_required, int *out) const {
197  int found_leaves[n_trees];
198 
199  /*
200  * The following loops over all trees, and routes the query to exactly one
201  * leaf in each.
202  */
203  #pragma omp parallel for
204  for (int n_tree = 0; n_tree < n_trees; ++n_tree) {
205  int idx_tree = 0;
206  for (int d = 0; d < depth; ++d) {
207  const int j = n_tree * depth + d;
208  const int idx_left = 2 * idx_tree + 1;
209  const int idx_right = idx_left + 1;
210  const float split_point = split_points(idx_tree, n_tree);
211  if (projected_query(j) <= split_point) {
212  idx_tree = idx_left;
213  } else {
214  idx_tree = idx_right;
215  }
216  }
217  found_leaves[n_tree] = idx_tree - (1 << depth) + 1;
218  }
219 
220  int n_elected = 0;
221  VectorXi votes = VectorXi::Zero(n_samples);
222 
223  // count votes
224  for (int n_tree = 0; n_tree < n_trees; ++n_tree) {
225  const VectorXi &idx_one_tree = tree_leaves[n_tree][found_leaves[n_tree]];
226  const int nn = idx_one_tree.size(), *data = idx_one_tree.data();
227  for (int i = 0; i < nn; ++i, ++data) {
228  if (++votes(*data) == votes_required) {
229  out[n_elected++] = *data;
230  }
231  }
232  }
233 
234  if(!test) {
235  if (n_elected < k) {
236  /*
237  * If not enough samples had at least votes_required
238  * votes, find the maximum amount of votes needed such
239  * that the final search set size has at least k samples
240  */
241  VectorXf::Index max_index;
242  votes.maxCoeff(&max_index);
243  int max_votes = votes(max_index);
244 
245  VectorXi vote_count = VectorXi::Zero(max_votes + 1);
246  for (int i = 0; i < n_samples; ++i)
247  vote_count(votes(i))++;
248 
249  for (int would_elect = 0; max_votes; --max_votes) {
250  would_elect += vote_count(max_votes);
251  if (would_elect >= k) break;
252  }
253 
254  for (int i = 0; i < n_samples; ++i) {
255  if (votes(i) >= max_votes && votes(i) < votes_required)
256  out[n_elected++] = i;
257  }
258  }
259  }
260 
261  return n_elected;
262  }

341  int exact_knn(const Map<VectorXf> &q, int k, const VectorXi &indices, int n_elected, int *out) const {
342  VectorXf distances(n_elected);
343 
344  if (sparse) {
345  #pragma omp parallel for
346  for (int i = 0; i < n_elected; ++i) {
347  distances(i) = X_norms(indices(i)) - 2 * Y->col(indices(i)).dot(q);
348  }
349  } else {
350  #pragma omp parallel for
351  for (int i = 0; i < n_elected; ++i) {
352  distances(i) = X_norms(indices(i)) - 2 * X->col(indices(i)).dot(q);
353  }
354  }
355 
356  if (k == 1) {
357  MatrixXf::Index index;
358  distances.minCoeff(&index);
359  out[0] = indices(index);
360  return 1;
361  }
362 
363  int mmin = std::min(k, n_elected);
364 
365  VectorXi idx(n_elected);
366  std::iota(idx.data(), idx.data() + n_elected, 0);
367  std::partial_sort(idx.data(), idx.data() + mmin, idx.data() + n_elected,
368  [&distances](int i1, int i2) {return distances(i1) < distances(i2);});
369 
370  for (int i = 0; i < mmin; ++i) out[i] = indices(idx(i));
371  return mmin;
372  }

513  std::vector<VectorXi> grow_subtree(const VectorXi &indices, int tree_level, int i, int n_tree, const MatrixXf &tree_projections) {
514  int n = indices.size();
515  int idx_left = 2 * i + 1;
516  int idx_right = idx_left + 1;
517 
518  if (tree_level == depth) {
519  std::vector<VectorXi> v;
520  v.push_back(indices);
521  return v;
522  }
523 
524  VectorXf projections(n);
525  for (int i = 0; i < n; ++i)
526  projections(i) = tree_projections(tree_level, indices(i));
527 
528  // sort indices of the projections based on their values
529  VectorXi ordered(n);
530  std::iota(ordered.data(), ordered.data() + n, 0);
531  std::sort(ordered.data(), ordered.data() + ordered.size(),
532  [&projections](int i1, int i2) {return projections(i1) < projections(i2);});
533 
534  int split_point = (n % 2) ? n / 2 : n / 2 - 1; // median split
535  int idx_split_point = ordered(split_point);
536  int idx_split_point2 = ordered(split_point + 1);
537 
538  split_points(i, n_tree) = (n % 2) ? projections(idx_split_point) :
539  (projections(idx_split_point) + projections(idx_split_point2)) / 2;
540  VectorXi left_indices = ordered.head(split_point + 1);
541  VectorXi right_indices = ordered.tail(n - split_point - 1);
542 
543  VectorXi left_elems = VectorXi(left_indices.size());
544  VectorXi right_elems = VectorXi(right_indices.size());
545 
546  for (int i = 0; i < left_indices.size(); ++i)
547  left_elems(i) = indices(left_indices(i));
548  for (int i = 0; i < right_indices.size(); ++i)
549  right_elems(i) = indices(right_indices(i));
550 
551  std::vector<VectorXi> v = grow_subtree(left_elems, tree_level + 1, idx_left, n_tree, tree_projections);
552  std::vector<VectorXi> w = grow_subtree(right_elems, tree_level + 1, idx_right, n_tree, tree_projections);
553  v.insert(v.end(), w.begin(), w.end());
554  return v;
555  }

566  void build_sparse_random_matrix() {
567  sparse_random_matrix = SparseMatrix<float, RowMajor>(n_pool, dim);
568 
569  std::random_device rd;
570  std::mt19937 gen(rd());
571  std::uniform_real_distribution<float> uni_dist(0, 1);
572  std::normal_distribution<float> norm_dist(0, 1);
573 
574  std::vector<Triplet<float>> triplets;
575  for (int j = 0; j < n_pool; ++j) {
576  for (int i = 0; i < dim; ++i) {
577  if (uni_dist(gen) > density) continue;
578  triplets.push_back(Triplet<float>(j, i, norm_dist(gen)));
579  }
580  }
581 
582  sparse_random_matrix.setFromTriplets(triplets.begin(), triplets.end());
583  sparse_random_matrix.makeCompressed();
584  }
